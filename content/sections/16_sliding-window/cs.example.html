// 슬라이딩 윈도우 기법을 이용해 길이 K인 부분 배열 중 최대 합을 구하는 예제 using System; class Program { static void Main() { int n = 6, k = 3; int[] arr = { 1, 3, 2, 6, 2, 8 }; // 초기 윈도우(첫 k개의 합)를 구함 int sum = 0; for (int i = 0; i < k; i++) { sum += arr[i]; } int max = sum; // 첫 윈도우의 합으로 초기화 // 윈도우를 한 칸씩 오른쪽으로 이동시키며 최대값 갱신 for (int i = k; i < n; i++) { sum += arr[i]; // 새로 들어온 값 추가 sum -= arr[i - k]; // 윈도우에서 빠져나간 값 제거 max = Math.Max(max, sum); // 최댓값 갱신 } // 결과 출력 Console.WriteLine(max); } }
