#include #include #include using namespace std; int main() { int n = 6, k = 3; vector arr = { 1, 3, 2, 6, 2, 8 }; // 초기 윈도우(첫 k개의 합)를 계산 int sum = 0; for (int i = 0; i < k; i++) { sum += arr[i]; } int maxSum = sum; // 첫 윈도우의 합으로 초기화 // 슬라이딩 윈도우를 한 칸씩 오른쪽으로 이동시키며 최대값 갱신 for (int i = k; i < n; i++) { sum += arr[i]; // 새로 들어온 값을 합에 추가 sum -= arr[i - k]; // 빠져나간 값을 합에서 제거 maxSum = max(maxSum, sum); // 현재 합과 최대값 비교 } // 결과 출력 cout << maxSum << endl; return 0; }
