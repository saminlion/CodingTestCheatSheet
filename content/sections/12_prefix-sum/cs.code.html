// 배열의 구간 합을 빠르게 계산하기 위한 누적 합(Prefix Sum) 기법 예제 int[] arr = { 1, 2, 3, 4, 5 }; // 원본 배열 (0번 인덱스부터 시작) int[] prefix = new int[arr.Length + 1]; // 누적 합 배열 (크기는 arr보다 1 더 큼) // prefix[i]에는 arr[0]부터 arr[i-1]까지의 합이 저장됨 // 예: prefix[1] = arr[0], prefix[2] = arr[0] + arr[1], ... for (int i = 1; i <= arr.Length; i++) { prefix[i] = prefix[i - 1] + arr[i - 1]; } // 예시: 구간 [2, 4]의 합 구하기 (0-based 인덱스 기준) // 구간 [2, 4]의 합은 arr[2] + arr[3] + arr[4] // = prefix[5] - prefix[2] // = (arr[0] + arr[1] + arr[2] + arr[3] + arr[4]) - (arr[0] + arr[1]) // = arr[2] + arr[3] + arr[4] int sum = prefix[4 + 1] - prefix[2]; // sum 결과는 3 + 4 + 5 = 12
