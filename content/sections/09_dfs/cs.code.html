// DFS(깊이 우선 탐색)를 이용한 그래프 탐색 // graph: 인접 리스트로 표현된 그래프 (정점 간 연결 관계) // visited: 각 노드의 방문 여부를 저장하는 배열 bool[] visited = new bool[n]; // 노드 방문 여부를 저장하는 배열 List[] graph = new List[n]; // 인접 리스트 그래프 초기화 for (int i = 0; i < n; i++) graph[i] = new(); // 각 리스트 인스턴스 할당 // DFS 함수 정의 void DFS(int node) { if (visited[node]) return; // 이미 방문한 노드는 건너뜀 visited[node] = true; // 현재 노드를 방문 처리 foreach (var next in graph[node]) { // 현재 노드에 연결된 이웃 노드 순회 DFS(next); // 재귀적으로 방문 } }
