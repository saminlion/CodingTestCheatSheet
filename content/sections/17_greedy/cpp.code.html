// 목적: 가장 큰 동전부터 차례대로 사용하여 동전 수를 최소화 // 알고리즘 유형: 그리디(Greedy) #include #include using namespace std; int main() { vector coins = {500, 100, 50, 10}; // 큰 단위부터 정렬된 동전 목록 int target = 1260; // 거슬러 줄 금액 int count = 0; // 동전 개수 누적용 for (int coin : coins) { count += target / coin; // 해당 동전으로 줄 수 있는 개수만큼 사용 target %= coin; // 남은 금액을 더 작은 동전으로 처리 // 예: 500원 2개 사용 → 남은 금액 260 // 100원 2개 사용 → 남은 금액 60 // ... } cout << count << endl; // 결과 출력: 최소 동전 개수 return 0; }
