// 목적: 큰 단위 동전부터 사용하여 거스름돈을 최소 개수로 구성 // 알고리즘 종류: 그리디(Greedy) int[] coins = {500, 100, 50, 10}; // 사용 가능한 동전 단위 (큰 값부터 정렬됨) int target = 1260; // 거슬러 줘야 할 총 금액 int count = 0; // 동전 개수를 누적할 변수 // 각 동전 단위를 하나씩 처리 foreach (int coin in coins) { count += target / coin; // 현재 동전으로 줄 수 있는 개수만큼 사용 target %= coin; // 남은 금액을 업데이트 // 예: 500원 2개 사용 → 남은 금액 260 // 100원 2개 사용 → 남은 금액 60 // ... } // count: 최소한의 동전 개수 Console.WriteLine(count);
