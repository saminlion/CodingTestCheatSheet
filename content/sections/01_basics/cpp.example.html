// 입력된 데이터 크기(N)에 따라 사용할 수 있는 알고리즘의 시간 복잡도를 안내하는 예제 // 입력: 정수 N // 출력: N에 적합한 알고리즘 복잡도 등급 안내 #include using namespace std; int main() { int N; cin >> N; if (N <= 1000) { cout << "O(N^2) 가능" << endl; } else if (N <= 1000000) { cout << "O(N log N) 이하만 가능" << endl; } else { cout << "O(N) 또는 O(1) 필요" << endl; } return 0; }
